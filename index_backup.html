Question 1: Swap two numbers without a third variable

#include <stdio.h>

int main() {
    int a = 10, b = 20;
    
    printf("Before swap: a = %d, b = %d\n", a, b);
    
    // Swap logic
    a = a + b;
    b = a - b;
    a = a - b;
    
    printf("After swap: a = %d, b = %d\n", a, b);
    
    return 0;
}

Explanation:

This program swaps two numbers without using a third variable by using arithmetic operations:

  1. a = a + b - Store sum in a
  2. b = a - b - Get original a value in b
  3. a = a - b - Get original b value in a

Question 2: Check if a number is Armstrong

#include <stdio.h>
#include <math.h>

int main() {
    int num, originalNum, remainder, result = 0, n = 0;
    
    printf("Enter an integer: ");
    scanf("%d", &num);
    
    originalNum = num;
    
    // Count number of digits
    while (originalNum != 0) {
        originalNum /= 10;
        ++n;
    }
    
    originalNum = num;
    
    // Calculate sum of nth power of digits
    while (originalNum != 0) {
        remainder = originalNum % 10;
        result += pow(remainder, n);
        originalNum /= 10;
    }
    
    if (result == num)
        printf("%d is an Armstrong number.\n", num);
    else
        printf("%d is not an Armstrong number.\n", num);
    
    return 0;
}

Explanation:

An Armstrong number is a number that is equal to the sum of its own digits each raised to the power of the number of digits.

Example: 153 = 1Â³ + 5Â³ + 3Â³ = 1 + 125 + 27 = 153

Question 3: Recursive function to find factorial

#include <stdio.h>

int factorial(int n) {
    // Base case
    if (n == 0 || n == 1)
        return 1;
    
    // Recursive case
    return n * factorial(n - 1);
}

int main() {
    int num;
    
    printf("Enter a positive integer: ");
    scanf("%d", &num);
    
    if (num < 0)
        printf("Factorial is not defined for negative numbers.\n");
    else
        printf("Factorial of %d = %d\n", num, factorial(num));
    
    return 0;
}

Explanation:

This recursive function calculates factorial by:

  1. Base case: If n is 0 or 1, return 1
  2. Recursive case: Return n Ã— factorial(n-1)

Example: factorial(5) = 5 Ã— 4 Ã— 3 Ã— 2 Ã— 1 = 120

int main() {

Question 5: Check if string is palindrome (without string.h)

#include <stdio.h>

int main() {
    char str[100];
    int i, length = 0, flag = 1;
    
    printf("Enter a string: ");
    scanf("%s", str);
    
    // Find length
    while (str[length] != '\0') {
        length++;
    }
    
    // Check palindrome
    for (i = 0; i < length / 2; i++) {
        if (str[i] != str[length - i - 1]) {
            flag = 0;
            break;
        }
    }
    
    if (flag)
        printf("%s is a palindrome.\n", str);
    else
        printf("%s is not a palindrome.\n", str);
    
    return 0;
}

Explanation:

A palindrome reads the same forwards and backwards (e.g., "madam", "racecar").

The program:

  1. Calculates string length without strlen()
  2. Compares characters from both ends moving towards center
  3. If any mismatch found, it's not a palindrome

Question 6: File Handling - Read and Write Even/Odd Numbers

#include <stdio.h>

int main() {
    FILE *input, *even, *odd;
    int num;
    
    // Open files
    input = fopen("numbers.txt", "r");
    even = fopen("EVEN.txt", "w");
    odd = fopen("ODD.txt", "w");
    
    if (input == NULL || even == NULL || odd == NULL) {
        printf("Error opening files!\n");
        return 1;
    }
    
    // Read numbers and write to respective files
    while (fscanf(input, "%d", &num) != EOF) {
        if (num % 2 == 0)
            fprintf(even, "%d\n", num);
        else
            fprintf(odd, "%d\n", num);
    }
    
    // Close files
    fclose(input);
    fclose(even);
    fclose(odd);
    
    printf("Numbers separated successfully!\n");
    
    return 0;
}

Explanation:

This program demonstrates file handling by:

  1. Opening input file in read mode ("r")
  2. Opening output files in write mode ("w")
  3. Reading numbers using fscanf()
  4. Writing to appropriate file using fprintf()
  5. Closing all files using fclose()

Question 8: Difference between Structure and Union

#include <stdio.h>

// Structure
struct Student {
    int roll;
    char name[20];
    float marks;
};

// Union
union Data {
    int i;
    float f;
    char str[20];
};

int main() {
    struct Student s1;
    union Data d1;
    
    printf("Size of structure: %lu bytes\n", sizeof(s1));
    printf("Size of union: %lu bytes\n\n", sizeof(d1));
    
    // Structure - all members have separate memory
    s1.roll = 101;
    strcpy(s1.name, "John");
    s1.marks = 85.5;
    printf("Structure values:\n");
    printf("Roll: %d, Name: %s, Marks: %.2f\n\n", s1.roll, s1.name, s1.marks);
    
    // Union - all members share same memory
    d1.i = 10;
    printf("Union after storing int: %d\n", d1.i);
    
    d1.f = 220.5;
    printf("Union after storing float: %.2f\n", d1.f);
    printf("Previous int value is lost: %d\n", d1.i);
    
    return 0;
}

Key Differences:

Feature Structure Union
Memory Each member has separate memory All members share same memory
Size Sum of all members Size of largest member
Access All members accessible anytime Only one member at a time
